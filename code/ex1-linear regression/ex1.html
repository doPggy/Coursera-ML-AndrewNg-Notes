函数自定义区

function cost = cost_func(x, y, theta, m)
% x .* y 对应元素相乘
h = x * theta;
c = (h - y) .^ 2;
cost = sum(c) / (2 * m);
end
多注意一下矩阵形状。
% 梯度下降，拟合出参数
function theta = gd(x, y, iterations, alpha)
theta = zeros(2, 1);
m = length(y);
for i = 1 : iterations
h = x * theta;
cost = sum((h - y) .* x, 1); % 沿着列加和，这是第一维度
theta = theta - alpha * cost.' ./ m; % 不转置会发生广播
end
end
